	AntAFKButton.MouseButton1Click:Connect(function()
		afkActive = not afkActive
		local bb = game:GetService('VirtualUser')
		local afkFunc
		if afkActive then
			AntiAFKFrameColor.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
			afkFunc = game:GetService('Players').LocalPlayer.Idled:Connect(function()
				bb:CaptureController()
				bb:ClickButton2(Vector2.new())
			end)
		else
			AntiAFKFrameColor.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
			afkFunc:Disconnect()
			afkFunc = nil
		end
	end)

	StatusButton.MouseButton1Click:Connect(function()

		loadstring(game:HttpGet("https://raw.githubusercontent.com/LDSHUBscripts/scripts/main/Status"))()
		wait(2)
		StatusFrameColor.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
	end)

	FPSButton.MouseButton1Click:Connect(function()
		local decalsyeeted = true
		local g = game
		local w = g.Workspace
		local l = g.Lighting
		local t = w.Terrain
		t.WaterWaveSize = 0
		t.WaterWaveSpeed = 0
		t.WaterReflectance = 0
		t.WaterTransparency = 0
		l.GlobalShadows = false
		l.FogEnd = 9e9
		l.Brightness = 0
		settings().Rendering.QualityLevel = "Level01"
		for i, v in pairs(g:GetDescendants()) do
			if v:IsA("Part") or v:IsA("Union") or v:IsA("CornerWedgePart") or v:IsA("TrussPart") then
				v.Material = "Plastic"
				v.Reflectance = 0
			elseif v:IsA("Decal") or v:IsA("Texture") and decalsyeeted then
				v.Transparency = 1
			elseif v:IsA("ParticleEmitter") or v:IsA("Trail") then
				v.Lifetime = NumberRange.new(0)
			elseif v:IsA("Explosion") then
				v.BlastPressure = 1
				v.BlastRadius = 1
			elseif v:IsA("Fire") or v:IsA("SpotLight") or v:IsA("Smoke") or v:IsA("Sparkles") then
				v.Enabled = false
			elseif v:IsA("MeshPart") then
				v.Material = "Plastic"
				v.Reflectance = 0
				v.TextureID = 10385902758728957
			end
		end
		for i, e in pairs(l:GetChildren()) do
			if e:IsA("BlurEffect") or e:IsA("SunRaysEffect") or e:IsA("ColorCorrectionEffect") or e:IsA("BloomEffect") or e:IsA("DepthOfFieldEffect") then
				e.Enabled = false
			end

local function offMenus()
	for _,v in pairs(Frame:GetChildren()) do
		if v:IsA("Frame") and v.Name ~= "Warn" and v.Name ~= "LE" then
			v.Visible = false
		end
	end
	for _,v in pairs(GButtons:GetChildren()) do
		if v:IsA("TextButton") then
			v.BackgroundColor3 = Color3.fromRGB(126, 126, 126)
		end
	end

local NameSave = game.Players.LocalPlayer.Name.."LDS"

local function LoadSettings()
	local HttpService = game:service('HttpService')

	--use 'isfile' to check if we are reading the correct txt file

	if isfile(NameSave) then
		--use 'readfile' to read 'TUTORIAL.txt' and decode it
		G.Settings = HttpService:JSONDecode(readfile(NameSave))
	end
end

local function SaveSettings()
	local JSON -- is nil
	local HttpService = game:service('HttpService') -- used for the JSON DECODE AND ENCODE
	--Decode
	JSON = HttpService:JSONEncode(G.Settings)
	--Make a text file using 'writefile'
	writefile(NameSave,JSON)
	-- write a code: check if the file was saved: yes - return true - no - return false. 
	if isfile(NameSave) then
		return true
	else
		return false
	end
end

					changeVisual()
				else
					value = false
				end

				if description ~= '' then
					helpIcon.MouseEnter:Connect(function()
						cloneFrame.HelpFrame.HelpText.Text = description
						cloneFrame.HelpFrame.Visible = true
					end)
					helpIcon.MouseLeave:Connect(function()
						cloneFrame.HelpFrame.Visible = false
					end)
				end
				togBt.MouseButton1Click:Connect(function()
					local execute = true
					if typeof(premium) == "boolean" and premium and not isPremium then
						execute = false
					end
					if execute then
						value = not value
						G.Settings[text] = value
						changeVisual()
						if typeof(premium) == "function" then
							premium(value)
						elseif typeof(premium) == "boolean" then
							callback(value)
						end
					else
						module:OkWarn("This is a premium feature!")
					end
				end)
			end

			function folder:addClick(text, description, size, premium, callback)
				local clickBt, helpIcon = createClickButton(text, description, cloneScroll)
				if size == "small" then
					clickBt.Parent.Size = UDim2.new(0.839, 0,0.058, 0)
				end
				if typeof(premium) == "boolean" and premium then
					createPremiumBg(clickBt.Parent)
					helpIcon.ImageColor3 = Color3.fromRGB(52, 52, 52)
					clickBt.Parent.BackgroundColor3 = Color3.fromRGB(186, 107, 255)
				end
				
				if description ~= '' then
					helpIcon.MouseEnter:Connect(function()
						cloneFrame.HelpFrame.HelpText.Text = description
						cloneFrame.HelpFrame.Visible = true
					end)
					helpIcon.MouseLeave:Connect(function()
						cloneFrame.HelpFrame.Visible = false
					end)
				end
				clickBt.MouseButton1Click:Connect(function()
					local execute = true

					if execute then
						if typeof(premium) == "function" then
							premium()
						elseif typeof(premium) == "boolean" then
							callback()
						end
					end
				end)	
			end

			function folder:addCombo(text, description, opts)
				local selectBt, helpIcon = createComboButton(text, description, cloneScroll)
				updateCanvasSize(cloneScroll, cloneScroll.UIListLayout)
				local value
				if table.find(G.Settings, text) then
					value = G.Settings[text]
					selectBt.Text = value
				else
					value = ""
				end
				if description ~= '' then
					helpIcon.MouseEnter:Connect(function()
						cloneFrame.HelpFrame.HelpText.Text = description
						cloneFrame.HelpFrame.Visible = true
					end)
					helpIcon.MouseLeave:Connect(function()
						cloneFrame.HelpFrame.Visible = false
					end)
				end
				selectBt.MouseButton1Click:Connect(function()
					tab.MenuFrame.ScrollingFrame.Visible = false
					local scroll = tab.OptionFrame.ScrollableOpt
					local baseOption = scroll.Option
					tab.OptionFrame.ComboTitle.Text = text

					-- Funções da gui

					local searchBox = tab.OptionFrame.SearchBox
					local searchF
					searchF = searchBox:GetPropertyChangedSignal("Text"):Connect(function()
						for _, v in pairs(scroll:GetChildren()) do
							if v.Name == "NewOption" then
								v:Destroy()
							end
						end

						local search = searchBox.Text:lower()
						for i, v in pairs(opts) do
							if string.sub(v:lower(), 1, #search) == search then
								local newOpt = baseOption:Clone()
								newOpt.Name = "NewOption"
								newOpt.Text = v
								newOpt.Parent = scroll
								newOpt.Visible = true

								newOpt.MouseButton1Click:Connect(function()
									value = v
									G.Settings[text] = v
									selectBt.Text = tostring(v)
									tab.OptionFrame.Visible = false
									tab.MenuFrame.ScrollingFrame.Visible = true
									for _, v in pairs(scroll:GetChildren()) do
										if v.Name == "NewOption" then
											v:Destroy()
										end
									end
									searchBox.Text = ""
									searchF:Disconnect()
								end)
							end
						end
					end)

					local closeF
					closeF = tab.OptionFrame.Close.MouseButton1Click:Connect(function()
						tab.OptionFrame.Visible = false
						for _, v in pairs(scroll:GetChildren()) do
							if v.Name == "NewOption" then
								v:Destroy()
							end
						end
						tab.MenuFrame.ScrollingFrame.Visible = true
						searchBox.Text = ""
						searchF:Disconnect()
						closeF:Disconnect()
					end)

					--  Funções / Opções
					for _, v in pairs(opts) do
						
						local clone = baseOption:Clone()
						clone.Parent = scroll
						clone.Visible = true
						clone.Name = "NewOption"
						clone.Text = v
						clone.MouseButton1Click:Connect(function()
							value = v
							selectBt.Text = tostring(v)
							tab.OptionFrame.Visible = false
							tab.MenuFrame.ScrollingFrame.Visible = true
							for _, v in pairs(scroll:GetChildren()) do
								if v.Name == "NewOption" then
									v:Destroy()
								end
							end
							searchBox.Text = ""
							searchF:Disconnect()
						end)
					end
					updateCanvasSize(scroll, scroll.UIGridLayout)
					tab.OptionFrame.Visible = true
				end)
				local combo = {}
				function combo:getValue()
					return value
				end
				return combo
			end

			function tab:addSlider(text, description, min, max, callback)
				local mouse = game.Players.LocalPlayer:GetMouse()
				local movingSlider = false
				local SliderClick, DescIcon, sliderBG, SliderValue = createSlider(text, description, cloneScroll)
				updateCanvasSize(cloneScroll, cloneScroll.UIListLayout)
				local value = min
				
				if description ~= '' then
					DescIcon.MouseEnter:Connect(function()
						cloneFrame.HelpFrame.HelpText.Text = description
						cloneFrame.HelpFrame.Visible = true
					end)
					DescIcon.MouseLeave:Connect(function()
						cloneFrame.HelpFrame.Visible = false
					end)
				end
				-- muda a pos da agulha
				SliderClick.MouseButton1Down:Connect(function()
					movingSlider = true
				end)
				Frame.MouseButton1Up:Connect(function()
					movingSlider = false
				end)
				UIS.InputEnded:Connect(function(input, gameProcessedEvent)
					if not gameProcessedEvent and input.UserInputType == Enum.UserInputType.MouseButton1 then
						movingSlider = false
					end
				end)

				local snapAmount = 1
				local pixelsFromEdge = 0

				mouse.Move:Connect(function()
					if movingSlider then

						local xOffset = math.floor((mouse.X - sliderBG.AbsolutePosition.X) / snapAmount + 0.5) * snapAmount
						local xOffsetClamped = math.clamp(xOffset, pixelsFromEdge, sliderBG.AbsoluteSize.X - pixelsFromEdge)

						local sliderPosNew = UDim2.new(0, xOffsetClamped, 0, SliderClick.Position.Y)

						SliderClick.Position = sliderPosNew

						local roundedAbsSize = math.floor(sliderBG.AbsoluteSize.X / snapAmount + 0.5) * snapAmount
						local roundedOffsetClamped = math.floor(xOffsetClamped / snapAmount + 0.5) * snapAmount

						local sliderValue = roundedOffsetClamped / roundedAbsSize

						value = math.round(sliderValue * max)
						if value < min then
							value = min
						end
						SliderValue.Text = value
						if callback ~= nil then
							callback(value)
						end
					end
				end)

				local sliderReturn = {}

				function sliderReturn:getValue()
					return value
				end

				return sliderReturn
			end

			function tab:addInputBox(text)
				local TextBox = createInputBox(text, cloneScroll)
				local textboxM = {}
				function textboxM:getValue()
					return TextBox.Text
				end
				return textboxM
			end

			return folder
		end

		return tab
	end

	return window
